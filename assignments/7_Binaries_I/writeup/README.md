int main()
{
    int a;
    int b;
    b = 0x1ceb00da;
    a = 0xfeedface;
    
    printf("a = %d\n", a);
    printf("b = %d\n", b);
    
    a = a ^ b;
    b = b ^ a;
    a = a ^ b;
    
    printf("a = %d\n", a);
    printf("b = %d\n", b);
    
    return 0;
}
